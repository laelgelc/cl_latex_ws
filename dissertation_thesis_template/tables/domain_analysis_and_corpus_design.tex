                \item Acknowledgements

    \textbf{Sampling methods}
    \begin{itemize}
        \item Balanced sample across
        \begin{itemize}
            \item Academic discipline
            \item Section
        \end{itemize}
        \item Procedure for the academic discipline stratum
        \begin{enumerate}
            \item Rank the disciplines by the number of paragraphs
            \item Balance the corpus according to the discipline that:
            \begin{itemize}
                \item contributes the least with paragraphs
                \item maximises the corpus' size
                \item causes the least reduction in the number of disciplines
            \end{itemize}
            \item Exclude the disciplines that are below the discipline of the previous step
        \end{enumerate}
        \item Procedure for the section stratum
        \begin{enumerate}
            \item Identify the patterns of sections that are common to all of the disciplines
            \item Exclude the diverging patterns
        \end{enumerate}
    \end{itemize}
