\begin{equation}
    \label{eq:sample_size_formula_ptbr}
    n=\frac{s^2}{(\frac{.5*CI_{range}}{t})^2}
\end{equation}
\noindent Equação: Fórmula de tamanho de amostra

, onde:
\begin{quote}
    $n$ = tamanho requerido da amostra (tamanho recomendado do corpus) \\
    $s$ = desvio padrão ($s^2$ = variância) \\
    $CI_{range}$ = alcance do intervalo de confiança \\
    $t$ = $t$-value de Student
\end{quote}
